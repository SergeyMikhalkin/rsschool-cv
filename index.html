                public static bool CalcViscosity(double viscosity1, double quantity1, double viscosity2, double quantity2, out double

                {
                    result = 0;
                
                    try
                    {

                    double b = quantity2 * Math.Log10(Math.Log10(viscosity2 + 0.8d));

                    c = Math.Pow(10.0d, c);
                    c -= 0.8d;
                
                    result = c;
                    }
                    catch (Exception)
                    {
                    return false;
                }
                
                return true;
                }
